int sumar(int a, int b) {
    int resultado = a + b;
    return resultado;
}

void main() {
    int x = 5;
    int y = 10;
    int z;

    z = sumar(x, y);

    if (z > 10) {
        z = z - 1;
    } else {
        z = z + 1;
    }

    for (int i = 0; i < 5; i = i + 1) {
        z = z + i;
    }
}
